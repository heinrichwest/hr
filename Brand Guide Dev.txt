5) Brand Provider — src/brand/BrandProvider.tsx

Use this to ensure the global stylesheet is loaded once, and to expose tokens via context if needed.

import React, { createContext, useContext, PropsWithChildren } from "react";
import { speccon, type SpecconTokens } from "./tokens";
import "./global.css";

const BrandContext = createContext<SpecconTokens>(speccon);

export function BrandProvider({ children }: PropsWithChildren) {
  return <BrandContext.Provider value={speccon}>{children}</BrandContext.Provider>;
}

export function useBrand() {
  return useContext(BrandContext);
}


Usage:

import { BrandProvider } from "./brand/BrandProvider";

export function App() {
  return (
    <BrandProvider>
      {/* routes/layout */}
    </BrandProvider>
  );
}

import React from "react";
import "./button.css";

type Variant = "primary" | "secondary" | "accent";

export function Button(
  props: React.ButtonHTMLAttributes<HTMLButtonElement> & { variant?: Variant }
) {
  const { variant = "primary", className = "", ...rest } = props;
  return <button className={`sc-btn sc-btn--${variant} ${className}`} {...rest} />;
}

Logo component — src/components/Logo/SpecconLogo.tsx

Enforce “no transforms / no recolour” by design: only allow choosing the asset variant.

import React from "react";

type Variant = "standard" | "reversed";

export function SpecconLogo({ variant = "standard", alt = "SpecCon" }: { variant?: Variant; alt?: string }) {
  const src = variant === "reversed"
    ? "/assets/brand/speccon-logo-reversed.svg"
    : "/assets/brand/speccon-logo-standard.svg";

  return <img src={src} alt={alt} style={{ display: "block", height: 32, width: "auto" }} />;
}


(Logo must never be tilted, recoloured, rearranged, etc.)
